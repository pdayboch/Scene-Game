def play()
	next_scene = @Start_scene
	while next_scene!='win' && next_scene != 'death'
		scene = @Map_layout.get_scene(next_scene)
		scene.set_profile(@Profile)
		scene.enter()
		scene.actions()
		next_scene = scene.choice()
	end

	@Map_layout.get_scene(next_scene).enter()
end

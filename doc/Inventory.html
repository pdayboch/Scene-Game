def use_item()
	puts "Which item do you desire? Select the item number:"
	i = 1
	@Items.each do |item|
		desc = item[1]
		puts "#{i}: #{desc}"
		i += 1
	end
	print "> "
	item_num = $stdin.gets.chomp.to_i
	
	item_selected = @Items[item_num - 1][1]
	
	return item_selected
end
